Q10)Min and Max

Sol: 
        function arrayMin(arr) {
            var min = arr[0];
            for(let i=0;i< arr.length;i++)
            if(min > arr[i+1]){
              min = arr[i+1];
            }
          return min;
          }
          
          function arrayMax(arr) {
             var max = arr[0];
            for(let j=0;j< arr.length;j++)
            if(max < arr[j+1]){
              max = arr[j+1];
            }
          return max;
          }

Q11)Birthday Game

Sol: 
function Birthday_Game(arr,D ,M)
    {
        var ways=0;
         for(let i=0; i < arr.length-M+1; i++){
           let sum=0;
           for(let j=i; j < M+i; j++){
             sum+=arr[j];
           }
           if (sum === D){
             ways++;
           }
         }
        return ways;
    }
